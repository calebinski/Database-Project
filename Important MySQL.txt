-------Table Creation---------
CREATE TABLE User (
    user_id INT,
    username VARCHAR (128),
    password VARCHAR(128),
    PRIMARY KEY(user_id)
);

CREATE TABLE Team (
    team_id INT,
    Name VARCHAR(128),
    PRIMARY KEY(team_id)
);

CREATE TABLE TeamMembers_Users (
    user_id INT,
    team_id INT,
    FOREIGN KEY (user_id) REFERENCES User(user_id),
    FOREIGN KEY (team_id) REFERENCES Team(team_id)
);

CREATE TABLE TeamLeader(
    user_id INT,
    team_id INT,
    FOREIGN KEY (user_id) REFERENCES User(user_id),
    FOREIGN KEY (team_id) REFERENCES Team(team_id)
);

CREATE TABLE Player_Character(
    name VARCHAR(128),
    class VARCHAR(128),
    level INT,
    intellect INT,
    strength INT,
    agility INT,
    stamina INT,
    armor INT,
    magic_resist INT,
    PRIMARY KEY(name)
);

CREATE TABLE TeamMembers_Characters (
    name VARCHAR(128),
    team_id INT,
    FOREIGN KEY (name) REFERENCES Player_Character(name),
    FOREIGN KEY (team_id) REFERENCES Team(team_id)
);

CREATE TABLE Gear (
    gear_id INT,
    name VARCHAR(128),
    type VARCHAR(128),
    intellect INT,
    strength INT,
    agility INT,
    stamina INT,
    armor INT,
    magic_resist INT,
    PRIMARY KEY(gear_id)
);

CREATE TABLE CharacterGear(
    name VARCHAR(128),
    gear_id INT,
    FOREIGN KEY (name) REFERENCES Player_Character(name),
    FOREIGN KEY (gear_id) REFERENCES Gear(gear_id)
);

CREATE TABLE Enemy (
    enemy_id INT,
    name VARCHAR(128),
    type VARCHAR(128),
    intellect INT,
    strength INT,
    agility INT,
    stamina INT,
    armor INT,
    magic_resist INT,
    PRIMARY KEY(enemy_id)
);

CREATE PROCEDURE dbapp.getcharacterbyuser(IN username VARCHAR(128))
BEGIN
    SELECT * 
    FROM player_character 
    INNER JOIN user_characters ON player_character.name = user_characters.name
    WHERE user_characters.user_id = username;
END //
